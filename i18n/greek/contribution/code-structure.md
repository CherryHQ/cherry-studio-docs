# Δομή Κώδικα

```yaml
...
├─ docs/ # Κατάλογος Τεκμηρίωσης
├─ resources/ # Κατάλογος Αρχείων Πόρων
├─ scripts/ # Κατάλογος Αρχείων Σεναρίων
└─ src/ # Κύριος Κατάλογος Πηγαίου Κώδικα
    ├─main/ # Κώδικας Κύριας Διεργασίας
    ├─preload/ # Κατάλογος Προ-φόρτωσης Σεναρίων
    └─renderer/ # Κώδικας Διεργασίας Απόδοσης
        ├─src/ # Πηγαίος Κώδικας Διεργασίας Απόδοσης
            ├─assets/ # Αρχεία Πόρων
                ├─fonts/ # Αρχεία Γραμματοσειρών
                ├─images/ # Αρχεία Εικόνων (προφίλ, κλπ.)
                └─styles/ # Αρχεία Στυλ
            ├─components/ # Στοιχεία
            ├─config/ # Ρυθμίσεις
            ├─context/ # Περιβάλλον
            ├─databases/ # Σχετικά Αρχεία Βάσεων Δεδομένων
            ├─hooks/ # Προσαρμοσμένα Hooks
            ├─i18n/ # Αρχεία Διεθνοποίησης
            ├─pages/ # Αρχεία Σελίδων
            ├─providers/ # Ρυθμίσεις Σχετικές με Πάροχους
            ├─services/ # Αρχεία Υπηρεσιών
            ├─store/ # Αρχεία Διαχείρισης Κατάστασης
            ├─types/ # Αρχεία Ορισμού Τύπων TypeScript
            ├─utils/ # Συναρτήσεις Βοηθητικού Προγράμματος
            ...
        ...
    ...
...
```
