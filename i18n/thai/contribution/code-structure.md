# โครงสร้างโค้ด

```yaml
...
├─ docs/ #ไดเรกทอรีเอกสาร
├─ resources/ #ไดเรกทอรีไฟล์ทรัพยากร
├─ scripts/ #ไดเรกทอรีไฟล์สคริปต์
└─ src/ #ไดเรกทอรีซอร์สโค้ดหลัก
    ├─main/ #โค้ดโปรเซสหลัก
    ├─preload/ #ไดเรกทอรีสคริปต์พรีโหลด
    └─renderer/ #โค้ดเรนเดอร์โปรเซส
        ├─src/ #ซอร์สโค้ดเรนเดอร์โปรเซส
            ├─assets/ #ไฟล์ทรัพยากร
                ├─fonts/ #ไฟล์ฟอนต์
                ├─images/ #ไฟล์รูปภาพ (เช่น ภาพโปรไฟล์)
                └─styles/ #ไฟล์สไตล์
            ├─components/ #คอมโพเนนต์
            ├─config/ #การตั้งค่า
            ├─context/ #คอนเท็กซ์
            ├─databases/ #ไฟล์ฐานข้อมูล
            ├─hooks/ #Hooks ที่กำหนดเอง
            ├─i18n/ #ไฟล์สากล
            ├─pages/ #ไฟล์หน้า
            ├─providers/ #การตั้งค่าผู้ให้บริการ
            ├─services/ #ไฟล์บริการ
            ├─store/ #ไฟล์จัดการสถานะ
            ├─types/ #ไฟล์นิยามประเภท TypeScript
            ├─utils/ #ฟังก์ชันยูทิลิตี้
            ...
        ...
    ...
...

```
