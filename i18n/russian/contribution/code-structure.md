# Структура кода

```yaml
...
├─ docs/ # Каталог документации
├─ resources/ # Каталог файлов ресурсов
├─ scripts/ # Каталог скриптов
└─ src/ # Основной каталог исходного кода
    ├─main/ # Код основного процесса
    ├─preload/ # Каталог скриптов предзагрузки
    └─renderer/ # Код рендер-процесса
        ├─src/ # Исходный код рендер-процесса
            ├─assets/ # Файлы ресурсов
                ├─fonts/ # Файлы шрифтов
                ├─images/ # Изображения (аватары и др.)
                └─styles/ # Файлы стилей
            ├─components/ # Компоненты
            ├─config/ # Конфигурационные файлы
            ├─context/ # Контекст
            ├─databases/ # Файлы баз данных
            ├─hooks/ # Кастомные хуки
            ├─i18n/ # Файлы интернационализации
            ├─pages/ # Файлы страниц
            ├─providers/ # Конфигурация провайдеров
            ├─services/ # Сервисные файлы
            ├─store/ # Файлы управления состоянием
            ├─types/ # Файлы определений типов TypeScript
            ├─utils/ # Вспомогательные функции
            ...
        ...
    ...
...
```
